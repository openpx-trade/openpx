Returns `(events, next_cursor)`.

## Parameters

<ParamField path="cursor" type="string?" />

<ParamField path="limit" type="integer?" />

<ParamField path="min_close_ts" type="i64?">
  Unix seconds — only events closing at or after this timestamp.
</ParamField>

<ParamField path="min_updated_ts" type="i64?">
  Unix seconds — only events updated at or after this timestamp.
</ParamField>

<ParamField path="series_id" type="string?">
  Filter by series ID / ticker.
</ParamField>

<ParamField path="status" type="string?">
  Filter by lifecycle status — venue-specific values (e.g. `open`, `closed`, `settled` on Kalshi; `closed=true/false` on Polymarket).
</ParamField>

<ParamField path="with_nested_markets" type="boolean?">
  Include the events' nested `Market` objects when supported.
</ParamField>

## Returns

<ResponseField name="events" type="Event[]">
  One page of `Event` results.

  <Expandable title="Event">
    <ResponseField name="category" type="string?" />
    <ResponseField name="description" type="string?" />
    <ResponseField name="end_ts" type="DateTime?" />
    <ResponseField name="id" type="string" />
    <ResponseField name="last_updated_ts" type="DateTime?" />
    <ResponseField name="market_ids" type="string[]?" />
    <ResponseField name="mutually_exclusive" type="boolean?" />
    <ResponseField name="open_interest" type="f64?" />

    <ResponseField name="start_ts" type="DateTime?" />
    <ResponseField name="status" type="string?" />
    <ResponseField name="title" type="string" />
    <ResponseField name="volume" type="f64?" />
  </Expandable>
</ResponseField>
<ResponseField name="cursor" type="string?">Opaque pagination cursor. `null` on the last page.</ResponseField>
