Returns `(snapshots, next_cursor)` for pagination.

## Parameters

<ParamField path="cursor" type="string?" />

<ParamField path="end_ts" type="i64?" />

<ParamField path="limit" type="integer?" />

<ParamField path="market_id" type="string" required />

<ParamField path="start_ts" type="i64?" />

<ParamField path="token_id" type="string?" />

## Returns

<ResponseField name="snapshots" type="OrderbookSnapshot[]">
  One page of `OrderbookSnapshot` results.

  <Expandable title="OrderbookSnapshot">
    <ResponseField name="asks" type="PriceLevel[]" />
    <ResponseField name="bids" type="PriceLevel[]" />
    <ResponseField name="hash" type="string?" />
    <ResponseField name="recorded_at" type="DateTime?" />
    <ResponseField name="timestamp" type="DateTime" />
  </Expandable>
</ResponseField>
<ResponseField name="cursor" type="string?">Opaque pagination cursor. `null` on the last page.</ResponseField>
