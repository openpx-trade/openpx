---
title: "fetch_trades"
sidebarTitle: "Fetch trades"
openapi: GET /v1/trades/fetch_trades
playground: simple
description: "Fetch recent public trades (\"tape\") for a market outcome."
---

Returns `(trades, next_cursor)` where `next_cursor` supports pagination.

## Parameters

<ParamField path="cursor" type="string?">
  Opaque pagination cursor from a previous response.
</ParamField>

<ParamField path="end_ts" type="i64?">
  Unix seconds (inclusive)
</ParamField>

<ParamField path="limit" type="integer?">
  Max number of trades to return (exchange-specific caps may apply).
</ParamField>

<ParamField path="market_id" type="string" required>
  Exchange-native market identifier.
</ParamField>

<ParamField path="market_ref" type="string?">
  Optional alternate market identifier for trade endpoints (e.g., Polymarket conditionId). When provided, exchanges should prefer this over `market_id`.
</ParamField>

<ParamField path="outcome" type="string?" />

<ParamField path="start_ts" type="i64?">
  Unix seconds (inclusive)
</ParamField>

<ParamField path="token_id" type="string?" />

## Returns

<ResponseField name="trades" type="MarketTrade[]">
  One page of `MarketTrade` results.

  <Expandable title="MarketTrade">
    <ResponseField name="aggressor_side" type="string?" />
    <ResponseField name="id" type="string?" />
    <ResponseField name="no_price" type="f64?" />
    <ResponseField name="outcome" type="string?" />
    <ResponseField name="price" type="f64" />
    <ResponseField name="side" type="string?" />
    <ResponseField name="size" type="f64" />
    <ResponseField name="source_channel" type="string" />
    <ResponseField name="taker_address" type="string?" />
    <ResponseField name="timestamp" type="DateTime" />
    <ResponseField name="tx_hash" type="string?" />
    <ResponseField name="yes_price" type="f64?" />
  </Expandable>
</ResponseField>
<ResponseField name="cursor" type="string?">Opaque pagination cursor. `null` on the last page.</ResponseField>
