    <ResponseField name="taker_fee_bps" type="f64?">Taker fee rate (basis points)</ResponseField>
    <ResponseField name="tick_size" type="f64?">Tick size (minimum price increment, normalized decimal e.g. 0.01)</ResponseField>
    <ResponseField name="title" type="string">Market title</ResponseField>
    <ResponseField name="token_id_no" type="string?">No outcome token ID</ResponseField>
