## Parameters

<ParamField path="condition_id" type="string?">
  Condition ID for OI enrichment (Polymarket).
</ParamField>

<ParamField path="end_ts" type="i64?">
  Unix seconds
</ParamField>

<ParamField path="interval" type="PriceHistoryInterval" required />

<ParamField path="market_id" type="string" required />

<ParamField path="outcome" type="string?" />

<ParamField path="start_ts" type="i64?">
  Unix seconds
</ParamField>

<ParamField path="token_id" type="string?" />

## Returns

<ResponseField name="result" type="Candlestick[]">
  List of `Candlestick` results.

  <Expandable title="Candlestick">
    <ResponseField name="close" type="f64" />
    <ResponseField name="high" type="f64" />
    <ResponseField name="low" type="f64" />
    <ResponseField name="open" type="f64" />
    <ResponseField name="open_interest" type="f64?">Open interest at this candle's close. Only available from exchanges that report it (e.g., Kalshi).</ResponseField>
    <ResponseField name="timestamp" type="DateTime">Period start timestamp (UTC). lightweight-charts expects start-of-period.</ResponseField>
    <ResponseField name="volume" type="f64">Trade volume in contracts. 0.0 if exchange doesn't provide volume.</ResponseField>
  </Expandable>
</ResponseField>
